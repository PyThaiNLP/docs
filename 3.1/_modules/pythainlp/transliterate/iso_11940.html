[docs]def transliterate(word: str) -> str:
    """
    Use ISO 11940 for transliteration
    :param str text: Thai text to be transliterated.
    :return: A string of IPA indicating how the text should be pronounced.
    """
    _new = ""
    for i in word:
        if i in _list_k:
            _new += _all_dict[i]
        else:
            _new += i
    return _new
